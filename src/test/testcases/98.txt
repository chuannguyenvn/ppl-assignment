                    isPrime: function boolean(n: integer) {
                        if (n <= 2) {
                            return false;
                        }
                        i: integer = 2;
                        for (i = 2, i * i < n, i + 1) {
                            if (n % i == 0) {
                                return false;
                            }
                        }
                        return true;
                    }
